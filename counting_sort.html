Java

        
public static void countingSort(int[] arr) {
  int max = Arrays.stream(arr).max().getAsInt();
  int[] count = new int[max + 1];
  for (int a : arr) count[a]++;
  int i = 0;
  for (int a = 0; a <= max; a++) {
    while (count[a]-- > 0) {
      arr[i++] = a;
    }
  }
}
        

      

      

        
C++

        
void countingSort(int arr[], int n) {
  int mx = *max_element(arr, arr + n);
  std::vector cnt(mx + 1, 0);
  for (int i = 0; i < n; i++) cnt[arr[i]]++;
  int idx = 0;
  for (int i = 0; i <= mx; i++) {
    while (cnt[i]--) {
      arr[idx++] = i;
    }
  }
}
        

      

    

    

      

        
Python

        
def counting_sort(arr):
    mx = max(arr)
    count = [0] * (mx + 1)
    for x in arr:
        count[x] += 1
    idx = 0
    for val, c in enumerate(count):
        for _ in range(c):
            arr[idx] = val
            idx += 1
        

      

      

        
JavaScript

        
function countingSort(arr) {
  let max = Math.max(...arr);
  let count = Array(max + 1).fill(0);
  arr.forEach(x => count[x]++);
  let i = 0;
  for (let x = 0; x <= max; x++) {
    while (count[x]--) {
      arr[i++] = x;
    }
  }
}
        

      

    

  






  
Counting Sort: Complexity & Key Points

  
    
    
  • Time complexity: O(n + k) where k is the value range (max element)
    • 
    
  • Space complexity: O(n + k)
    • 
    
  • Only for non-negative integers (or shift negative domain)
    • 
    
  • Non-comparison, integer sorting algorithm
    • 
    
  • Stable (with a version storing indices)
    • 
    
  • Optimal for small ranges; impractical if k ≫ n
    • 
    
  • Excellent as a building block for Radix Sort
