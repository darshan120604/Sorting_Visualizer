Java

          
  public static void bucketSort(int[] arr) {
    int n = arr.length;
    List<Integer>[] buckets = new ArrayList[10];
    for(int i = 0; i < 10; i++)
      buckets[i] = new ArrayList<>();
    for(int num : arr)
      buckets[num / 100].add(num);
    for(List<Integer> b : buckets)
      Collections.sort(b);
    int idx = 0;
    for(List<Integer> b : buckets)
      for(int v : b)
        arr[idx++] = v;
  }

  void bucketSort(int arr[], int n) {
    vector<int> buckets[10];
    for(int i = 0; i < n; i++)
      buckets[arr[i]/100].push_back(arr[i]);
    for(int i = 0; i < 10; i++)
      sort(buckets[i].begin(), buckets[i].end());
    int idx = 0;
    for(int i = 0; i < 10; i++)
      for(int v : buckets[i])
        arr[idx++] = v;
  }

  def bucket_sort(arr):
    buckets = [[] for _ in range(10)]
    for num in arr:
      buckets[num // 100].append(num)
    for bucket in buckets:
      bucket.sort()
    idx = 0
    for bucket in buckets:
      for num in bucket:
        arr[idx] = num
        idx += 1
          

        

        

          
JavaScript

          
  function bucketSort(arr) {
    const buckets = Array.from({length: 10}, () => []);
    arr.forEach(num => buckets[Math.floor(num / 100)].push(num));
    buckets.forEach(bucket => bucket.sort((a, b) => a - b));
    let i = 0;
    for (const bucket of buckets)
      for (const num of bucket)
        arr[i++] = num;
  }
          

        

      

    

  

  


  
Bucket Sort: Complexity & Key Points

  
    
    
  • Worst-case time complexity: O(n²) (if all elements go into one bucket)
    • 
    
  • Best-case/Average-case time complexity: O(n + k) (if elements are uniformly distributed among k buckets)
    • 
    
  • Space complexity: O(n + k) (buckets plus the input array)
    • 
    
  • Great for sorting uniformly distributed data over a known range
    • 
    
  • Divides data into buckets, sorts each (typically with insertion sort inside buckets)
    • 
    
  • Not in-place, uses extra storage for buckets
    • 
    
  • Visualization helps to see data distribution and bucket sorting
