    Pure Bubble Sort in Java / C++ / Python / JavaScript
    

      

        

          
Java


public static void bubbleSort(int[] arr) {
    int n = arr.length;
    for (int i = 0; i < n; i++) {
        boolean swapped = false;
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
                swapped = true;
            }
        }
        if (!swapped) break;
    }
}

        

        

          
C++


void bubbleSort(vector<int>& arr) {
    int n = arr.size();
    for (int i = 0; i < n; i++) {
        bool swapped = false;
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                swap(arr[j], arr[j + 1]);
                swapped = true;
            }
        }
        if (!swapped) break;
    }
}

def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        swapped = False
        for j in range(n - i - 1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
                swapped = True
        if not swapped:
            break


        

        

          
JavaScript


function bubbleSort(arr) {
  let n = arr.length;
  for (let i = 0; i < n; i++) {
    let swapped = false;
    for (let j = 0; j < n - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
        swapped = true;
      }
    }
    if (!swapped) break;
  }
}


        

      

    

  

  

    
Bubble Sort: Complexity & Key Points

    
    
      
  • Worst-case time complexity: O(n²) (when the array is reversed)
    • 
      
  • Best-case time complexity: O(n) (already sorted, with optimization!)
    • 
      
  • Average-case time complexity: O(n²)
    • 
      
  • Space complexity: O(1) (in-place, only a few variables)
    • 
    

    
    
      
  • Simple to code, but not efficient for large arrays.
    • 
      
  • After each pass, the largest unsorted value "bubbles" to the end.
    • 
      
  • With the optimization used here, the algorithm stops early if no swaps are needed in a pass.
    • 
      
  • Always a stable sorting method (equal elements retain their original order).
    • 
      
  • Good for teaching or visualizing sort logic—not used in performance applications.
